// ===== FILE: web/src/components/ui/Modal.tsx =====
import React, { forwardRef, useEffect, useCallback, useId, useRef } from 'react'
import { createPortal } from 'react-dom'
import { cva, cx, type VariantProps } from 'class-variance-authority'

const FOCUSABLE =
  'a[href],button:not([disabled]),input:not([disabled]),select:not([disabled]),textarea:not([disabled]),[tabindex]:not([tabindex="-1"])'

const dialogVariants = cva(
  [
    'bg-sage-content-bg border border-sage-border rounded-xl shadow-2xl',
    'flex flex-col w-full max-h-[calc(100vh-3rem)]',
    'animate-modal-dialog-in',
  ],
  {
    variants: {
      size: {
        sm: 'max-w-sm',
        md: 'max-w-xl',
        lg: 'max-w-3xl',
        xl: 'max-w-5xl',
      },
    },
    defaultVariants: { size: 'md' },
  }
)

type ModalSize    = NonNullable<VariantProps<typeof dialogVariants>['size']>
type ModalVariant = 'default' | 'alert'

interface ModalProps extends VariantProps<typeof dialogVariants> {
  open: boolean
  onClose: () => void
  title?: React.ReactNode
  variant?: ModalVariant
  closeOnOverlayClick?: boolean
  children?: React.ReactNode
  footer?: React.ReactNode
}

const Modal = forwardRef<HTMLDivElement, ModalProps>(
  (
    {
      open,
      onClose,
      title,
      size,
      variant = 'default',
      closeOnOverlayClick = true,
      children,
      footer,
    },
    forwardedRef
  ) => {
    const uid       = useId()
    const titleId   = title !== undefined ? `${uid}-title` : undefined
    const internalRef = useRef<HTMLDivElement>(null)

    const mergedRef = useCallback(
      (node: HTMLDivElement | null) => {
        ;(internalRef as React.MutableRefObject<HTMLDivElement | null>).current = node
        if (typeof forwardedRef === 'function') forwardedRef(node)
        else if (forwardedRef)
          (forwardedRef as React.MutableRefObject<HTMLDivElement | null>).current = node
      },
      [forwardedRef]
    )

    useEffect(() => {
      if (!open) return
      const dialog = internalRef.current
      if (!dialog) return

      const getFocusable = () =>
        Array.from(dialog.querySelectorAll<HTMLElement>(FOCUSABLE))

      const initial = getFocusable()
      if (initial.length > 0) initial[0].focus()
      else dialog.focus()

      const trap = (e: KeyboardEvent) => {
        if (e.key === 'Escape') { onClose(); return }
        if (e.key !== 'Tab') return
        const els   = getFocusable()
        if (els.length === 0) { e.preventDefault(); return }
        const first = els[0]
        const last  = els[els.length - 1]
        if (e.shiftKey) {
          if (document.activeElement === first || document.activeElement === dialog) {
            e.preventDefault()
            last.focus()
          }
        } else {
          if (document.activeElement === last) {
            e.preventDefault()
            first.focus()
          }
        }
      }

      document.addEventListener('keydown', trap)
      const prevOverflow = document.body.style.overflow
      document.body.style.overflow = 'hidden'
      return () => {
        document.removeEventListener('keydown', trap)
        document.body.style.overflow = prevOverflow
      }
    }, [open, onClose])

    if (!open || typeof document === 'undefined') return null

    return createPortal(
      <div
        className="fixed inset-0 z-[200] bg-sage-overlay backdrop-blur-sm flex items-center justify-center p-6 animate-modal-overlay-in"
        onClick={
          closeOnOverlayClick
            ? (e) => { if (e.target === e.currentTarget) onClose() }
            : undefined
        }
      >
        <div
          ref={mergedRef}
          role="dialog"
          aria-modal="true"
          aria-labelledby={titleId}
          tabIndex={-1}
          className={cx(dialogVariants({ size }))}
        >
          {title !== undefined && (
            <div className="flex items-center justify-between gap-3 px-6 py-[1.125rem] border-b border-sage-border flex-shrink-0">
              <h2
                id={titleId}
                className="text-[0.9375rem] font-semibold text-sage-text leading-snug m-0"
              >
                {title}
              </h2>
              <button
                type="button"
                onClick={onClose}
                aria-label="Close"
                className={cx(
                  'inline-flex items-center justify-center flex-shrink-0 p-0',
                  'w-7 h-7 rounded-md border border-transparent bg-transparent',
                  'text-sage-rail-text cursor-pointer transition-colors duration-[120ms]',
                  'hover:bg-sage-accent-light hover:text-sage-sidebar-active-text',
                  'hover:border-sage-risk-external-border',
                  'focus-visible:outline-none focus-visible:ring-2',
                  'focus-visible:ring-sage-accent focus-visible:ring-offset-2'
                )}
              >
                <svg width="12" height="12" viewBox="0 0 12 12" fill="none" aria-hidden="true">
                  <path
                    d="M1 1l10 10M11 1L1 11"
                    stroke="currentColor"
                    strokeWidth="1.75"
                    strokeLinecap="round"
                  />
                </svg>
              </button>
            </div>
          )}

          <div className="px-6 py-5 overflow-y-auto flex-1 text-sage-rail-text text-sm leading-relaxed">
            {children}
          </div>

          {footer && (
            <div className="flex items-center justify-end gap-2 px-6 py-4 border-t border-sage-border flex-shrink-0">
              {footer}
            </div>
          )}
        </div>
      </div>,
      document.body
    )
  }
)

Modal.displayName = 'Modal'

export { Modal }
export type { ModalProps, ModalSize, ModalVariant }

import React, { forwardRef } from 'react'
import { cva, cx, type VariantProps } from 'class-variance-authority'

const selectVariants = cva(
  [
    'appearance-none w-full font-sage rounded-md cursor-pointer',
    'bg-sage-content-bg text-sage-text leading-tight',
    'transition-[border-color,box-shadow,opacity] duration-150',
    'focus:outline-none',
    'disabled:opacity-50 disabled:cursor-not-allowed disabled:bg-sage-rail-bg',
  ],
  {
    variants: {
      variant: {
        default: [
          'border border-sage-border',
          'enabled:hover:border-sage-accent',
          'focus:border-sage-accent focus:ring-[3px] focus:ring-sage-accent-light focus:ring-offset-0',
        ],
        error: [
          'border border-sage-risk-destructive-border',
          'enabled:hover:border-sage-risk-destructive-fg',
          'focus:border-sage-risk-destructive-fg focus:ring-[3px] focus:ring-sage-risk-destructive-bg focus:ring-offset-0',
        ],
        success: [
          'border border-sage-risk-ephemeral-border',
          'enabled:hover:border-sage-risk-ephemeral-fg',
          'focus:border-sage-risk-ephemeral-fg focus:ring-[3px] focus:ring-sage-risk-ephemeral-bg focus:ring-offset-0',
        ],
      },
      size: {
        sm: 'text-xs  py-1            pl-[0.625rem] pr-8',
        md: 'text-sm  py-2            pl-[0.875rem] pr-9',
        lg: 'text-base py-[0.625rem]  pl-4          pr-10',
      },
    },
    defaultVariants: { variant: 'default', size: 'md' },
  }
)

type SelectVariant = NonNullable<VariantProps<typeof selectVariants>['variant']>
type SelectSize    = NonNullable<VariantProps<typeof selectVariants>['size']>

interface SelectOption {
  value: string
  label: string
  disabled?: boolean
}

interface SelectProps
  extends Omit<React.SelectHTMLAttributes<HTMLSelectElement>, 'size'>,
    VariantProps<typeof selectVariants> {
  options?: SelectOption[]
  placeholder?: string
}

const arrowRight: Record<SelectSize, string> = {
  sm: 'right-2',
  md: 'right-[0.625rem]',
  lg: 'right-3',
}

const arrowDim: Record<SelectSize, number> = { sm: 10, md: 12, lg: 14 }

const Select = forwardRef<HTMLSelectElement, SelectProps>(
  ({ variant, size = 'md', options, placeholder, className, children, ...rest }, ref) => {
    const s = size ?? 'md'
    return (
      <div className="relative inline-block w-full">
        <select
          ref={ref}
          className={cx(selectVariants({ variant, size }), className)}
          {...rest}
        >
          {placeholder && (
            <option value="" disabled>
              {placeholder}
            </option>
          )}
          {options
            ? options.map((opt) => (
                <option key={opt.value} value={opt.value} disabled={opt.disabled}>
                  {opt.label}
                </option>
              ))
            : children}
        </select>

        <span
          className={cx(
            'absolute top-1/2 -translate-y-1/2 pointer-events-none',
            'text-sage-rail-text flex items-center',
            arrowRight[s]
          )}
          aria-hidden="true"
        >
          <svg
            width={arrowDim[s]}
            height={arrowDim[s]}
            viewBox="0 0 12 12"
            fill="none"
          >
            <path
              d="M2 4.5l4 4 4-4"
              stroke="currentColor"
              strokeWidth="1.5"
              strokeLinecap="round"
              strokeLinejoin="round"
            />
          </svg>
        </span>
      </div>
    )
  }
)

Select.displayName = 'Select'

export { Select, selectVariants }
export type { SelectProps, SelectVariant, SelectSize, SelectOption }
