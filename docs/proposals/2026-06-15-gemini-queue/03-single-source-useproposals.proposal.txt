export function formatRelativeTime(iso: string): string {
  const ms = Date.now() - new Date(iso).getTime()
  if (ms < 60_000) return `${Math.floor(ms / 1000)}s ago`
  if (ms < 3_600_000) return `${Math.floor(ms / 60_000)}m ago`
  if (ms < 86_400_000) return `${Math.floor(ms / 3_600_000)}h ago`
  return `${Math.floor(ms / 86_400_000)}d ago`
}

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import {
  fetchProposals,
  approveProposal,
  rejectProposal,
  approveBatchProposals,
} from '../api/client'
import type { Proposal } from '../api/client'

export interface BatchApproveResult {
  results: { status: string; trace_id: string }[]
}

const PROPOSALS_KEY = ['proposals-pending'] as const

export function useProposals() {
  const queryClient = useQueryClient()

  const { data, isLoading, error } = useQuery({
    queryKey: PROPOSALS_KEY,
    queryFn: fetchProposals,
    refetchInterval: 5_000,
  })

  const proposals: Proposal[] = data?.proposals ?? []

  const approveMutation = useMutation({
    mutationFn: (vars: { traceId: string; approver: string; feedback: string }) =>
      approveProposal(vars.traceId, vars.approver, vars.feedback),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: PROPOSALS_KEY }),
  })

  const rejectMutation = useMutation({
    mutationFn: (vars: { traceId: string; approver: string; feedback: string }) =>
      rejectProposal(vars.traceId, vars.approver, vars.feedback),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: PROPOSALS_KEY }),
  })

  const batchApproveMutation = useMutation({
    mutationFn: (vars: { traceIds: string[]; approver: string }) =>
      approveBatchProposals(vars.traceIds, vars.approver),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: PROPOSALS_KEY }),
  })

  return {
    proposals,
    isLoading,
    error,
    data,
    approve: (traceId: string, approver: string, feedback: string) =>
      approveMutation.mutateAsync({ traceId, approver, feedback }),
    rejectFull: (traceId: string, approver: string, feedback: string) =>
      rejectMutation.mutateAsync({ traceId, approver, feedback }),
    batchApprove: (traceIds: string[], approver: string): Promise<BatchApproveResult> =>
      batchApproveMutation.mutateAsync({ traceIds, approver }),
    isBatchPending: batchApproveMutation.isPending,
  }
}

function formatCountdown(iso: string): string {
  const ms = new Date(iso).getTime() - Date.now()
  if (ms <= 0) return 'Expired'
  if (ms < 60_000) return `${Math.floor(ms / 1000)}s`
  if (ms < 3_600_000) return `${Math.floor(ms / 60_000)}m`
  return `${Math.floor(ms / 3_600_000)}h`
}

function formatActionType(actionType: string): string {
  return actionType
    .replace(/_/g, ' ')
    .replace(/\b\w/g, c => c.toUpperCase())
}

function sortProposals(proposals: Proposal[]): Proposal[] {
  return [...proposals].sort((a, b) => {
    const ai = RISK_ORDER.indexOf(a.risk_class)
    const bi = RISK_ORDER.indexOf(b.risk_class)
    if (ai !== bi) return ai - bi
    return new Date(b.created_at).getTime() - new Date(a.created_at).getTime()
  })
}

    if (selectedId) setSeenIds(prev => new Set([...prev, selectedId]))
  }, [selectedId])

  const addToast = useCallback((message: string, type: 'success' | 'error') => {
    const id = Math.random().toString(36).slice(2)
    setToasts(prev => [...prev, { id, message, type }])
    setTimeout(() => setToasts(prev => prev.filter(t => t.id !== id)), 4000)
  }, [])

  const removeProposal = useCallback(
    (traceId: string) => {
      setCheckedIds(prev => {
        const s = new Set(prev)
        s.delete(traceId)
        return s
      })
      const idx = sorted.findIndex(p => p.trace_id === traceId)
      const next = sorted[idx + 1] ?? sorted[idx - 1]
      setSelectedId(next?.trace_id ?? null)
    },
    [sorted],
  )

  const handleApprove = useCallback(
    async (traceId: string, approver: string, feedback: string) => {
      try {
        await approve(traceId, approver || 'human', feedback)
        addToast('Proposal approved', 'success')
        removeProposal(traceId)
      } catch {
        addToast('Approval failed — check API', 'error')
      }
    },
    [approve, addToast, removeProposal],
  )

  const handleReject = useCallback(
    async (traceId: string, approver: string, feedback: string) => {
      try {
        await rejectFull(traceId, approver || 'human', feedback)
        addToast('Proposal rejected', 'success')
        removeProposal(traceId)
      } catch {
        addToast('Rejection failed — check API', 'error')
      }
    },
    [rejectFull, addToast, removeProposal],
  )

  const undoMutation = useMutation({
    mutationFn: (trace_id: string) => undoProposal(trace_id),
    onSuccess: () => addToast('Undo triggered — changes will be reverted', 'success'),
    onError: (err: Error) => addToast(err.message, 'error'),
  })

  const handleBatchApprove = useCallback(async () => {
    try {
      const result = await batchApprove(
        [...checkedIds],
        localStorage.getItem('sage_approver_name') || 'human',
      )
      const count = result.results.filter(
        (r: { status: string }) => r.status === 'approved',
      ).length
      addToast(`Approved ${count} proposal${count !== 1 ? 's' : ''}`, 'success')
      setCheckedIds(new Set())
    } catch {
      addToast('Batch approval failed', 'error')
    }
  }, [batchApprove, checkedIds, addToast])

  const checkedList = sorted.filter(p => checkedIds.has(p.trace_id))
  const allLowRisk = checkedList.every(
    p => p.risk_class === 'INFORMATIONAL' || p.risk_class === 'EPHEMERAL',
  )
  const showCheckboxes = sorted.length > 1
  const secondsAgo = Math.floor((Date.now() - lastUpdated.getTime()) / 1000)
  const selected = sorted.find(p => p.trace_id === selectedId) ?? null

  // suppress unused-variable warning for approvalRolesData used by callers
  void approvalRolesData

  return (
    <div
      className="flex h-full overflow-hidden"
      style={{ backgroundColor: '#fafafa', margin: '-24px' }}
    >
      <div
        className="flex flex-col shrink-0 overflow-hidden"
        style={{
          width: '320px',
          backgroundColor: '#ffffff',
          borderRight: '1px solid #e5e7eb',
        }}
      >
        <div
          className="px-4 py-3 shrink-0 flex items-center gap-2"
          style={{ borderBottom: '1px solid #e5e7eb' }}
        >
          <Inbox size={15} style={{ color: '#9ca3af' }} />
          <span className="text-sm font-semibold flex-1" style={{ color: '#374151' }}>
            Pending Approvals
          </span>
          {pendingCount > 0 && (
            <span
              className="text-xs font-bold px-2 py-0.5"
              style={{ backgroundColor: '#ef4444', color: '#fff' }}
            >
              {pendingCount}
            </span>
          )}
        </div>

        {showCheckboxes && (
          <div
            className="px-3 py-2 flex items-center gap-2 shrink-0"
            style={{ borderBottom: '1px solid #e5e7eb', backgroundColor: '#fafafa' }}
          >
            <input
              type="checkbox"
              checked={checkedIds.size === sorted.length && sorted.length > 0}
              onChange={e => {
                if (e.target.checked) setCheckedIds(new Set(sorted.map(p => p.trace_id)))
                else setCheckedIds(new Set())
              }}
              style={{ accentColor: '#f4f4f5' }}
            />
            <span className="text-xs flex-1" style={{ color: '#9ca3af' }}>
              {checkedIds.size > 0 ? `${checkedIds.size} selected` : 'Select all'}
            </span>
            {checkedIds.size > 0 && allLowRisk && (
              <button
                onClick={handleBatchApprove}
                disabled={isBatchPending}
                className="flex items-center gap-1 text-xs px-2 py-1 font-medium"
                style={{ backgroundColor: '#16a34a', color: '#fff' }}
              >
                <CheckSquare size={11} />
                Approve {checkedIds.size}
              </button>
            )}
            {checkedIds.size > 0 && !allLowRisk && (
              <span className="text-[10px]" style={{ color: '#9ca3af' }}>
                High-risk selected
              </span>
            )}
          </div>
        )}

        <div className="flex-1 overflow-y-auto">
          {isLoading && (
            <div
              className="flex items-center justify-center h-32 gap-2"
              style={{ color: '#9ca3af' }}
            >
              <RefreshCw size={14} className="animate-spin" />
              <span className="text-xs">Loading…</span>
            </div>
          )}
          {!isLoading && sorted.length === 0 && (
            <div className="flex flex-col items-center justify-center h-48 px-6 text-center">
              <CheckSquare size={32} style={{ color: '#d1d5db' }} />
              <p className="mt-3 text-sm font-medium" style={{ color: '#9ca3af' }}>
                No pending approvals
              </p>
              <p className="mt-1 text-xs" style={{ color: '#d1d5db' }}>
                Your agents are waiting for new tasks
              </p>
            </div>
          )}
          {sorted.map(p => (
            <ProposalRow
              key={p.trace_id}
              proposal={p}
              selected={selectedId === p.trace_id}
              isNew={!seenIds.has(p.trace_id)}
              onClick={() => setSelectedId(p.trace_id)}
              checked={checkedIds.has(p.trace_id)}
              onCheck={v =>
                setCheckedIds(prev => {
                  const s = new Set(prev)
                  if (v) s.add(p.trace_id)
                  else s.delete(p.trace_id)
                  return s
                })
              }
              showCheckbox={showCheckboxes}
            />
          ))}
        </div>

        <div
          className="px-3 py-2 flex items-center gap-1.5 shrink-0"
          style={{ borderTop: '1px solid #e5e7eb' }}
        >
          <RefreshCw size={10} style={{ color: '#d1d5db' }} />
          <span className="text-[10px]" style={{ color: '#d1d5db' }}>
            Updated {secondsAgo}s ago
          </span>
        </div>
      </div>

      <DetailPanel
        proposal={selected}
        onApprove={handleApprove}
        onReject={handleReject}
        onUndo={traceId => undoMutation.mutate(traceId)}
        loading={isBatchPending}
      />

      <ToastContainer
        toasts={toasts}
        onDismiss={id => setToasts(prev => prev.filter(t => t.id !== id))}
      />
    </div>
  )
}
